```javascript
const PENDING = "pending"; // 等待（默认）
const FULFILLED = "fulfilled"; // 成功
const REJECTED = "rejected"; // 失败

class MyPromise {
  // new MyPromise((resolve, reject) => {})
  constructor(executor) {
    // 默认是等待状态
    this.status = PENDING;
    // 成功的值
    this.value = undefined;
    // 成功的回调
    this.onResolvedCallbacks = [];
    // 成功
    const reslove = (value) => {
      // 状态改变之后，就不能再改变了
      if (this.status === PENDING) {
        this.value = value;
        this.status = FULFILLED;
        // 执行对应的成功回调
        this.onResolvedCallbacks.forEach((cb) => cb());
      }
    };
    // 失败的原因
    this.reason = undefined;
    // 失败的回调
    this.onRejectedCallbacks = [];
    // 失败
    const reject = (reason) => {
      // 状态改变之后，就不能再改变了
      if (this.status === PENDING) {
        this.reason = reason;
        this.status = REJECTED;
        // 执行对应的失败回调
        this.onRejectedCallbacks.forEach((cb) => cb());
      }
    };
    try {
      // executor执行器会立刻执行
      executor(reslove, reject);
    } catch (error) {
      // executor执行器内部报错，则reject
      reject(error);
    }
  }
  // 多次调用then方法，会按照调用顺序执行多次
  // 每次执行完then，都会返回一个新的Promise，通过调用新Promise的resolve或reject传值
  // then返回普通值，将这个值作为下一个then成功的值
  // then返回Promise，会执行这个Promise，并采用它执行后的状态，执行下一个then的成功或失败回调
  // then链式调用中，通过返回一个Pending状态的Promise，可以终止下面的then的执行
  // then链式调用中，中间的then失败了，也会继续执行后面的then
  then(onFulFilled, onRejected) {
    // onFulFilled和onRejected是可选参数
    onFulFilled =
      typeof onFulFilled === "function" ? onFulFilled : (data) => data;
    onRejected =
      typeof onRejected === "function"
        ? onRejected
        : (err) => { throw err };
    const promise2 = new MyPromise((reslove, reject) => {
      // 这样在执行then的时候，能拿到新Promise的reslove和reject
      if (this.status === FULFILLED) {
        queueMicrotask(() => {
          try {
            // x可能是普通值或Promise，或则执行onFulFilled直接报错
            let x = onFulFilled(this.value);
            // 判断x的值类型，但是此时拿不到promise2，需要异步
            reslovePromise(promise2, x, reslove, reject);
          } catch (error) {
            reject(error);
          }
        });
      }
      if (this.status === REJECTED) {
        queueMicrotask(() => {
          try {
            let x = onRejected(this.reason);
            reslovePromise(promise2, x, reslove, reject);
          } catch (error) {
            reject(error);
          }
        });
      }
      // 如果Promise中的代码是异步的，需要先订阅对应的then回调
      if (this.status === PENDING) {
        this.onResolvedCallbacks.push(() => {
          queueMicrotask(() => {
            try {
              let x = onFulFilled(this.value);
              reslovePromise(promise2, x, reslove, reject);
            } catch (error) {
              reject(error);
            }
          });
        });
        this.onRejectedCallbacks.push(() => {
          queueMicrotask(() => {
            try {
              let x = onRejected(this.reason);
              reslovePromise(promise2, x, reslove, reject);
            } catch (error) {
              reject(error);
            }
          });
        });
      }
    });
    return promise2;
  }
  catch() {}
  static reject(reason) {
    return new MyPromise((_, reject) => reject(reason))
  }
  // data为普通值时，直接成功，data为Promise时，等待其完成后再成功
  static resolve(data) {
    return new MyPromise((resolve) => {
      if (data instanceof MyPromise) {
        data.then(resolve);
      }
      // then返回普通值，将这个值作为下一个then成功的值
      else {
        resolve(data);
      }
    });
  }
  // 返回一个Promise，等待promises都成功后状态才变为成功
  static all(promises) {
    return new MyPromise((resolve, reject) => {
      const result = [];
      const length = promises.length;
      let count = 0;
      function processData(key, value) {
        result[key] = value;
        count++;
        // 每一项都执行完了才resolve
        if (count === length) {
          resolve(result);
        }
      }
      for (let i = 0; i < promises.length; i++) {
        const current = promises[i];
        if (current instanceof MyPromise) {
          // 其中一个Promise失败了，整个Promise就失败了
          current.then((data) => {
            processData(i, data);
          }, reject);
        } else {
          // 普通项直接返回
          processData(i, current);
        }
      }
    });
  }
  // 返回一个Promise，其中一个promises成功后，状态就变为成功
  static race(promises) {
    return new MyPromise((resolve, reject) => {
      for (let i = 0; i < promises.length; i++) {
        const current = promises[i];
        // 利用了Promise状态一旦改变,就不会再变的特性
        if (current instanceof MyPromise) {
          // 其中一个Promise成功了整个就成功
          current.then(resolve, reject);
        } else {
          // 普通项直接成功
          resolve(current);
        }
      }
    });
  }
}
```

```javascript
function debounce(fn, delay) {
  let timer = null;
  return function () {
    timer && clearTimeout(timer);
    timer = setTimeout(() => {
      fn.call(this, ...arguments);
    }, delay);
  };
}
```

#### 深拷贝

```javascript
// 引用类型
const ObjectType = "[object Object]";
const ArrayType = "[object Array]";
const MapType = "[object Map]";
const SetType = "[object Set]";
const DateType = "[object Date]";
const RegExpType = "[object RegExp]";
// 可遍历的引用类型
const iteratorTypes = {
  [ObjectType]: true,
  [ArrayType]: true,
  [MapType]: true,
  [SetType]: true,
};
function cloneDeep(target = {}, map = new WeakMap()) {
  // null 原始类型 函数 => 直接返回
  if (target === null || typeof target !== "object") {
    return target;
  }
  const extractType = Object.prototype.toString.call(target);
  // RegExp Date
  if (extractType === RegExpType || extractType === DateType) {
    return new target.constructor(target);
  }
  // 循环引用
  if (map.has(target)) {
    return map.get(target);
  }
  let cloneTarget;
  // 可遍历的类型Object|Array|Map|Set|WeakMap|WeakSet，调用其构造函数进行初始化
  if (iteratorTypes[extractType]) {
    cloneTarget = new target.constructor();
  }
  map.set(target, cloneTarget);
  // Map 
  if (extractType === MapType) {
    target.forEach((value, key) => cloneTarget.set(key, value));
    return cloneTarget;
  }
  // Set 
  if (extractType === SetType) {
    target.forEach((value) => cloneTarget.add(value));
    return cloneTarget;
  }
  // 判断是数组
  const isArray = extractType === ArrayType;
  const targetKeys = isArray ? target : Object.keys(target);
  // 使用for循环代替for in循环
  for (let i = 0, len = targetKeys.length; i < len; i++) {
    // 数组使用下标、对象使用key
    const key = isArray ? i : targetKeys[i];
    cloneTarget[key] = cloneDeep(target[key], map);
  }
  // for (let key in target) {
  //   cloneTarget[key] = cloneDeep(target[key], map);
  // }
  return cloneTarget;
}
```

```javascript
function curry(fn, ...args) {
  // 需要柯里化的函数的入参个数（不包括默认参数）
  const length = fn.length;
  return function () {
    // 拼接每次的入参
    const innerArgs = [...args, ...arguments];
    // 参数数量不够则递归调用，继续拼接
    if (innerArgs.length < length) {
      return curry.call(this, fn, ...innerArgs);
    }
    // 参数数量满足，调用函数返回结果
    return fn.apply(this, innerArgs);
  };
}
```

```javascript
function bubbleSort(arr = []) {
    const length = arr.length
    // 外层循环只需要length-1次
    for (let i = 0; i < length - 1; i ++) {
        // 外层循环只需要length-1-i次，排过的数不需要再排
        for(let j = 0; j < length - 1 - i; j ++) {
            if (arr[j] > arr[j + 1]) {
                swap(arr, j, j + 1)
            }
        }
    }
    return arr
}
```

#### 插入排序

```javascript
// 1. 把数组中的第一个元素A看作一个有序列表[A]
// 2. 遍历后面的元素，依次与有序列表从后往前比较，涉及到插入和交换位置
// [2] 1 5 3 6 4
// 1与[2]依次比较，1<2，2往后移动一位，遍历结束，1插入原来2的位置 [1 2] 5 3 6 4
// 5与[1 2]依次比较，5>2，直接插入 [1 2 5] 3 6 4
// 3与[1 2 5]依次比较，3<5，5往后移动一位，3>2，3插入原来5的位置 [1 2 3 5] 6 4
// ...
function insertSort(arr = []) {
    let i,j
    for (i = 1; i < arr.length; i ++) {
        // 当前需要比较的元素cur
        const cur = arr[i]
        // 当前有序列表的长度i
        for (j = i - 1; j >= 0 && cur < arr[j]; j --) {
            // 当cur比有序列表中的元素B小时，则需要将B往后移动一位（cur已经保存了下一位的值）
            arr[j + 1] = arr[j]
        }
        // cur需要插入的位置j+1
        arr[j + 1] = cur
    }
    return arr
}
```

#### 快速排序

```javascript
// 1. 从数组中选择一个基准值pivot
// 2. 比pivot小的元素全部放左边，比pivot大的元素全部放右边
// 3. 得到左右两个新序列，分别再递归执行1、2，直到新序列中只有一个元素

// 4 5 1 3 7 8 2 9 6
// 执行partition
// pivot = 4, low = 0, high = 8
// 1. 从右往左找比pivot小的数，找到2，j=6
// 2. 从左往右找比pivot大的数，找到5，i=1
// 3. i<j则交换i和j上的元素 4 (2) 1 3 7 8 (5) 9 6
// 继续遍历...
// 执行1.找到3，j=3
// 执行2.直到i=j时也没找到比pivot大的数，i=3
// 结束遍历...
// 此时交换low和i上的元素 (3) 2 1 (4) 7 8 5 9 6
// 这时以i为分割点，左边的数都比pivot小，右边的数都比pivot大
// 递归执行...
function partition(arr, low, high) {
    // 取第一个元素作为基准值
    const pivot = arr[low]
    let i = low + 1
    let j = high
    while(i < j) {
        // 注意先右后左，因为基准值是第一个元素，右边默认是比它大的
        // 从右往左寻找，找到第一个比pivot小的值B
        while(i < j && arr[j] >= pivot) {
            j --
        }
        // 从左到右寻找，找到第一个比pivot大的值A
        while(i < j && arr[i] <= pivot) {
            i ++
        }
        // 只找到A或B时，i=j，此时这轮遍历结束
        // A和B都找到时，i<j，直接交换arr[i]和arr[j]，然后继续遍历
        if (i < j) {
            swap(arr, i, j)
        }
    }
    // 交换low和i上的元素
    swap(arr, low, i)
    // 返回i为下次递归的基准值位置做参考
    return i
}
function quickSort(arr = [], low = 0, high = arr.length - 1) {
    if (low < high) {
        const pivotKey = partition(arr, low, high)
        // 递归执行左边的序列
        quickSort(arr, low, pivotKey - 1)
        // 递归执行右边的序列
        quickSort(arr, pivotKey + 1, high)
    }
    return arr
}
```

```javascript
function binarySearch(arr = [], target) {
    let low = 0
    let high = arr.length - 1
    let mid
    // 这边需要考虑等于的情况
    while(low <= high) {
        mid = low + ((high - low) >> 1)
        if (arr[mid] === target) {
            return mid
        }
        if (arr[mid] > target) {
            high = mid - 1
        } else if (arr[mid] < target) {
            low = mid + 1
        }
    }
}
```

```javascript
// 中左右
function preorderTraversal(root) {
    const stack = []
    const result = []
    // 1.根节点先入栈
	stack.push(root)
    // 2.遍历栈
    while(stack.length) {
        // 3.依次取出栈顶元素
        const node = stack.pop()
        // 访问元素
        result.push(node)
        // 4.右子节点先入栈
        if (node.right) {
            stack.push(node.right)
        }
        // 5.左子节点后入栈
        if (node.left) {
            stack.push(node.left)
        }
    }
    return result
}
```

```javascript
// 左中右
function inorderTraversal(root) {
    const stack = []
    const result = []
    let cur = root
    while(stack.length || cur) {
        // 1.循环将当前节点和其所有右子节点入栈
        while (cur) {

            cur = cur.left

    }
```

```javascript
function postorderTraversal(root) {
    const stack = []
    const result = []
    // 1.根节点先入栈
	stack.push(root)
    // 2.遍历栈
    while(stack.length) {
        // 3.依次取出栈顶元素
        const node = stack.pop()
        // 前插法访问元素
        result.unshift(node)

            stack.push(node.left)
        }
        // 5.右子节点先入栈
        if (node.right) {
            stack.push(node.right)
        }
    }
    return result
}
```
